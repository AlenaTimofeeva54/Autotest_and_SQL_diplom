Задача №1.

SELECT c.login,
       COUNT(o."inDelivery") AS vdostavke
FROM "Couriers" AS c
INNER JOIN "Orders" AS o ON o."courierId" = c.ID
WHERE o."inDelivery" = 'true'
  AND o.finished = 'f'
GROUP BY c.id;

Задача №2.

SELECT track,
       (CASE
            WHEN finished = 't' THEN '2'
            WHEN cancelled = 't' THEN '-1'
            WHEN "inDelivery" = 't' THEN '1'
            ELSE '0'
        END) AS status
FROM "Orders";
